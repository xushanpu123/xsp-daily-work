## 实验1：

#### 

#### 1、捕获LoadFault

```rust
match scause.cause() {

    // 断点中断（ebreak）

    Trap::Exception(Exception::Breakpoint) => breakpoint(context),

    // 时钟中断

    Trap::Interrupt(Interrupt::SupervisorTimer) => supervisor_timer(context),

    Trap::Exception(Exception::LoadFault) => panic!(),





    // 其他情况，终止当前线程

    _ => fault(context, scause, stval),

  };
```

如上述代码所示，插入

```rust
Trap::Exception(Exception::LoadFault) => panic!(),
```

即可。

#### 

#### 2、在处理异常的过程中，如果程序想要非法访问的地址是 `0x0`，则打印 `SUCCESS!`。

match内内容改为：

```
Trap::Exception(Exception::LoadFault) => response_loadfault(stval),
```

并给出response_loadfault的定义：

```rust
fn response_loadfault(stval: usize){

  if stval == 0 as usize{

    println!("Success!");

  }

}
```

#### 

#### 3、添加或修改少量代码，使得运行时触发这个异常，并且打印出 `SUCCESS!`。

在breakpoint函数中，将context.sepc设为了0.

```rust
context.sepc = 0;
```

此后，由于跳出中断的地址始终为0，所以程序开始无限输出Success！



## 遇到的问题和解决方案：

为了解决此处的问题，本人陷入了一些思考，因为触发ebreak中断，程序被强制跳到了0处而无法跳回，所以需要一个参数来记录返回地址，因此本人采取了类似于C语言中全局变量的unsafe编程，修改部分如下：

​      

```rust
static mut restore : usize = 0;
```



```rust
fn breakpoint(context: &mut Context) {
    println!("Breakpoint at 0x{:x}", context.sepc);
    unsafe{
        restore = context.sepc;    //此处先保存需要恢复的地址
    }
    context.sepc = 0;
   
}
```

```rust
fn response_loadfault(context:&mut Context,stval: usize){ //此处需要恢复context，所以需要加入context参数
    if stval == 0 as usize{
        println!("Success!");
        context.sepc = restore;
    }
}
```

自然，interrupt_handle函数中关于地址访问错误的声明也应该加入context参数，此处不再赘叙。



## 实验二：此处主要记录线段树实现内存分配的实现，代码很丑陋，希望各位批评指正

```rust
use super::Allocator;

use alloc::{vec, vec::Vec};//提供线段树结构实现的分配器 [`Segment_Tree_Allocator`]



/// 使用线段树实现分配器

///

/// 在 `Vec` 末尾进行加入 / 删除。

/// 每个元素 tuple `(start, end)` 表示 [start, end) 区间为可用。



pub struct node{

  l:usize,

  r:usize,

  available:bool,

}

impl node {

  pub fn new(l:usize,r:usize)->Self{

​    Self{

​      l,

​      r,

​      available:false,

​    }

  }

}

pub struct SegmentTreeAllocator {

  tree: Vec<node>,

}

impl SegmentTreeAllocator{

  pub fn build(&mut self,u:usize,l:usize,r:usize){

​    self.tree[u]=node{l,r,available:true};

​    if l<r{

​      let mid:usize = (l+r)/2;

​      self.build(u*2, l, mid);

​      self.build(u*2+1,mid+1,r);

  }

}

  pub fn pushup(&mut self,u:usize){

​    self.tree[u].available=self.tree[u*2].available|self.tree[u*2+1].available;

  }

  pub fn modify(&mut self,u:usize,k:usize,d:bool){

​    let mid = (self.tree[u].l+self.tree[u].r)/2;

​    if k<=mid{

​      self.modify(u*2,k,d);

​    }

​    else{

​      self.modify(u*2+1,k,d);

​    }

​    self.pushup(u);

  }

  pub fn search(&self,u:usize)->Option<usize>{

​    if self.tree[u].available==false{

​      None

​    }

​    else{

​      if self.tree[u].l==self.tree[u].r{

​        Some(self.tree[u].l)

​      }

​      else{

​      if self.tree[u*2].available==true{

​        return self.search(u*2);

​      }

​      else{

​        return self.search(u*2+1);

​      }

​    }

​    }

  }

}



impl Allocator for SegmentTreeAllocator {

  fn new(capacity: usize) -> Self {

​    let mut tree = vec![node::new(1,capacity)];

​    for i in 1..(4*capacity){

​      tree.push(node::new(1,capacity));

​    }

​    let mut res = Self{tree};

​    res.build(1,1,capacity);

​    res

​    }

  



  fn alloc(&mut self) -> Option<usize> {

​    let res = self.search(1);

​    if let Some(x)=res{

​      self.modify(1,x,false);

​    }

​    res

  }



  fn dealloc(&mut self, index: usize) {

​    self.modify(1,index,true);

  }



}
```
